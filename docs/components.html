Toast notifications

Toast notifications can be created with the optional browser script and dismissed automatically or by the user.

HTML:

<p>
  <button type="button" onclick="LegacyCss.toast.show({ title: 'Saved', message: 'The request was updated.', type: 'success', position: 'top-right' })">Show toast</button>
</p>

JavaScript API:

<script>
  LegacyCss.toast.show({
    title: "Saved",
    message: "The request was updated.",
    type: "success",
    duration: 5000,
    position: "top-right"
  });

  LegacyCss.toast.close(toastElement);
  LegacyCss.toast.clear();
</script>

jQuery API:

<script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>
<script>
  $.toast({
    title: "Queued",
    message: "The export will run in the background.",
    type: "warning",
    position: "bottom-left"
  });

  $(".toast").toast("close");
</script>

Pagination

Use LegacyCss.pagination.setup() to load paged records into a target element. The example uses a local array, but the same callback shape can call a backend.

Paginated requests
Protocol Requester Status Updated

HTML:

<table>
  <caption>Paginated requests</caption>
  <thead>
    <tr>
      <th>Protocol</th>
      <th>Requester</th>
      <th>Status</th>
      <th>Updated</th>
    </tr>
  </thead>
  <tbody id="pagination-requests"></tbody>
</table>
<nav class="pagination" id="request-pagination" aria-label="Request pages"></nav>

JavaScript API with local data:

<script>
  const requests = [
    { protocol: "2026-00124", requester: "Ana Costa", status: "In review", updated: "06/13/2026" },
    { protocol: "2026-00125", requester: "Bruno Lima", status: "Pending", updated: "06/12/2026" }
  ];

  LegacyCss.pagination.setup("#request-pagination", {
    data: requests,
    target: "#pagination-requests",
    pageSize: 5,
    pageSizes: [5, 10, 25],
    renderItem(request) {
      const row = document.createElement("tr");

      row.innerHTML =
        "<td>" + request.protocol + "</td>" +
        "<td>" + request.requester + "</td>" +
        "<td>" + request.status + "</td>" +
        "<td>" + request.updated + "</td>";

      return row;
    }
  });
</script>

Backend-shaped loader:

<script>
  LegacyCss.pagination.setup("#request-pagination", {
    target: "#pagination-requests",
    pageSize: 10,
    load({ page, pageSize }) {
      return fetch("/api/requests?page=" + page + "&pageSize=" + pageSize)
        .then((response) => response.json());
    },
    renderItem(request) {
      return "<tr><td>" + request.protocol + "</td></tr>";
    }
  });
</script>

jQuery API:

<script>
  $("#request-pagination").pagination({
    data: requests,
    target: "#pagination-requests",
    pageSize: 5
  });
  $("#request-pagination").pagination("goTo", 2);
  $("#request-pagination").pagination("pageSize", 10);
  $("#request-pagination").pagination("refresh");
</script>
Multiselect

Use a native <select multiple> with data-multiselect for automatic progressive enhancement.

Selected values: ana, carla

HTML with automatic JavaScript behavior:

<label for="reviewers">Reviewers</label>
<select id="reviewers" name="reviewers[]" multiple data-multiselect data-placeholder="Choose reviewers">
  <option value="ana" selected>Ana Costa</option>
  <option value="bruno">Bruno Lima</option>
  <option value="carla" selected>Carla Mendes</option>
  <option value="diego">Diego Rocha</option>
  <option value="elisa" disabled>Elisa Souza</option>
</select>
<p id="reviewers-output" role="status">Selected values: ana, carla</p>
<p>
  <button type="button" id="reviewers-default">Select default reviewers</button>
  <button type="button" id="reviewers-clear">Clear reviewers</button>
</p>

JavaScript API:

<script>
  LegacyCss.multiselect.setup("#reviewers");
  LegacyCss.multiselect.open("#reviewers");
  LegacyCss.multiselect.close("#reviewers");
  LegacyCss.multiselect.toggle("#reviewers");

  const reviewers = document.getElementById("reviewers");
  const output = document.getElementById("reviewers-output");

  function getSelectedReviewers() {
    return Array.from(reviewers.selectedOptions).map((option) => option.value);
  }

  function setSelectedReviewers(values) {
    Array.from(reviewers.options).forEach((option) => {
      option.selected = values.includes(option.value);
    });

    reviewers.dispatchEvent(new Event("change", { bubbles: true }));
  }

  function renderSelectedReviewers() {
    const values = getSelectedReviewers();

    output.textContent = "Selected values: " + (values.join(", ") || "none");
  }

  reviewers.addEventListener("change", renderSelectedReviewers);
  document.getElementById("reviewers-default").addEventListener("click", () => {
    setSelectedReviewers(["ana", "carla"]);
  });
  document.getElementById("reviewers-clear").addEventListener("click", () => {
    setSelectedReviewers([]);
  });
  renderSelectedReviewers();
</script>

jQuery API:

<script>
  $("#reviewers").multiselect();
  $("#reviewers").multiselect("open");
  $("#reviewers").multiselect("close");
  $("#reviewers").multiselect("toggle");

  function getSelectedReviewers() {
    return $("#reviewers").val() || [];
  }

  function setSelectedReviewers(values) {
    $("#reviewers").val(values).trigger("change");
  }
</script>
